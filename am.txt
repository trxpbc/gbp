        function updateDateTime() {

        const pills = document.querySelectorAll('.pill');
        
        function setActivePill() {
            const params = new URLSearchParams(window.location.search);
            const currentPage = params.get('page'); // URL se 'page' value nikalta hai

            let matched = false;

            pills.forEach(pill => {
                // Pill text ko URL parameter se match karne ke liye clean up (space hatana etc)
                const pillType = pill.textContent.replace(/\s+/g, '').toLowerCase();
                
                if (currentPage && pillType === currentPage.toLowerCase()) {
                    pills.forEach(p => p.classList.remove('active'));
                    pill.classList.add('active');
                    pill.focus();
                    matched = true;
                }

                // Keyboard and Scroll Logic
                pill.addEventListener('focus', () => {
                    pill.scrollIntoView({ behavior: 'smooth', inline: 'center', block: 'nearest' });
                });
            });

            // Agar URL me kuch na ho ya galat ho, toh HOME (first pill) ko active rakhein
            if (!matched) {
                pills[0].classList.add('active');
                pills[0].focus();
            }
        }

        // --- 3. Keyboard Navigation Support ---
        document.addEventListener('keydown', (e) => {
            const active = document.activeElement;
            if (active.classList.contains('pill')) {
                if (e.key === 'ArrowRight') {
                    const next = active.nextElementSibling;
                    if (next) next.focus();
                }
                if (e.key === 'ArrowLeft') {
                    const prev = active.previousElementSibling;
                    if (prev) prev.focus();
                }
                if (e.key === 'Enter') {
                    // Enter dabane par page navigate karega
                    window.location.href = active.href;
                }
            }
        });

        // Run on load
        window.onload = setActivePill;
    </script>

    (function() {
        'use strict';

        // --- Configuration Constants ---
        const LOGIN_STATE_KEY = 'jtvp_login_state';
        const REQUIRED_STATE_VALUE = 'LOGGED_IN_TEST';
        const LOGIN_SOURCE_QUERY = 'm=1';
        const REDIRECT_URL = "https://com-jio-media-jiotvplus-ct.blogspot.com/?m=jio.php";
        const OVERLAY_ID = 'no-internet-overlay';
        const HIDDEN_CLASS = 'connection-hidden';

        let overlayElement; 

        /**
         * Handles the logic for checking login status and performing a redirect.
         */
        function handleAuthenticationAndRedirect() {
            const urlParams = new URLSearchParams(window.location.search);
            const urlStatus = urlParams.get('status'); 
            const storedState = localStorage.getItem(LOGIN_STATE_KEY);
            let shouldRedirect = false;
            
            // Check for URL trigger
            if (urlStatus && urlStatus.toUpperCase() === REQUIRED_STATE_VALUE.substring(0, 4)) {
                localStorage.setItem(LOGIN_STATE_KEY, REQUIRED_STATE_VALUE);
                shouldRedirect = true;
            } 
            // Check for stored state
            else if (storedState === REQUIRED_STATE_VALUE) {
                shouldRedirect = true;
            }

            if (shouldRedirect) {
                const currentURL = window.location.href;
                if (currentURL.includes(LOGIN_SOURCE_QUERY) && !currentURL.includes('m=jio.php')) {
                    window.location.replace(REDIRECT_URL);
                    console.log("JTVP: Redirecting after successful login state check.");
                }
            }
        }

        /**
         * Updates the visibility of the full-page overlay based on connection status.
         */
        function updateConnectionStatus() {
            if (!overlayElement) return;

            if (navigator.onLine) {
                // Hide the overlay using the transition class
                overlayElement.classList.add(HIDDEN_CLASS);
                console.log("JTVP: Connection Status - Online");
            } else {
                // Show the overlay
                overlayElement.classList.remove(HIDDEN_CLASS);
                console.log("JTVP: Connection Status - Offline");
            }
        }

        /**
         * Initializes the script, runs checks, and sets up event listeners.
         */
        function initialize() {
            overlayElement = document.getElementById(OVERLAY_ID);

            // 1. Authentication Logic
            handleAuthenticationAndRedirect();
            
            // 2. Offline Overlay Logic
            if (overlayElement) {
                updateConnectionStatus();
                window.addEventListener('online', updateConnectionStatus);
                window.addEventListener('offline', updateConnectionStatus);
            } else {
                console.error(`JTVP: Offline overlay element with ID '${OVERLAY_ID}' not found.`);
            }
        }

        window.addEventListener('DOMContentLoaded', initialize);
        
    })();
