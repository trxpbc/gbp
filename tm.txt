        lucide.createIcons();

        const pills = document.querySelectorAll('.pill');
        const searchInput = document.getElementById('searchInput');

        function setActivePill() {
            const currentUrl = window.location.href.toLowerCase();
            const savedPillId = localStorage.getItem('lastActivePill');
            let matched = false;

            pills.forEach(p => p.classList.remove('active'));

            for (let pill of pills) {
                const hrefPart = pill.getAttribute('href').toLowerCase();
                if (currentUrl.includes(hrefPart)) {
                    pill.classList.add('active');
                    pill.scrollIntoView({ behavior: 'smooth', inline: 'center' });
                    localStorage.setItem('lastActivePill', pill.id);
                    matched = true;
                    break;
                }
            }

            if (!matched && savedPillId) {
                const savedPill = document.getElementById(savedPillId);
                if (savedPill) {
                    savedPill.classList.add('active');
                    matched = true;
                }
            }
        }

        pills.forEach(pill => {
            pill.addEventListener('click', function() {
                localStorage.setItem('lastActivePill', this.id);
            });
            pill.addEventListener('focus', function() {
                this.scrollIntoView({ behavior: 'smooth', inline: 'center' });
            });
        });

        // Remote Key Handling
        document.addEventListener('keydown', (e) => {
            const active = document.activeElement;

            // 1. Arrow Right
            if (e.key === 'ArrowRight') {
                if (active.nextElementSibling && active.classList.contains('pill')) {
                    active.nextElementSibling.focus();
                } else if (active.classList.contains('pill')) {
                    searchInput.focus();
                }
            }
            
            // 2. Arrow Left
            if (e.key === 'ArrowLeft') {
                if (active === searchInput) {
                    pills[pills.length - 1].focus();
                } else if (active.previousElementSibling && active.classList.contains('pill')) {
                    active.previousElementSibling.focus();
                }
            }

            // 4. Enter Key
            if (e.key === 'Enter' && active.classList.contains('pill')) {
                active.click();
            }
        });

        // Fast Load Execution
        if (document.readyState === 'loading') {
            document.addEventListener('DOMContentLoaded', setActivePill);
        } else {
            setActivePill();
        }
    </script>

    function patchXHR(token) {
        const origOpen = XMLHttpRequest.prototype.open;
        XMLHttpRequest.prototype.open = function(method, url) {
            if (url && typeof url === 'string' && !url.includes("__hdnea__")) {
                url += (url.includes("?") ? "&" : "?") + "__hdnea__=" + token;
            }
            return origOpen.apply(this, arguments);
        };
    }

    async function loadPlaylist() {
        const params = new URLSearchParams(window.location.search);
        const m3uMain = params.get("cod") || localStorage.getItem("cod_url") || DEFAULT_PLAYLIST_URL;
        const srcLink = params.get("cookie") || params.get("src");

        try {
            // Updated Cookie logic to handle JSON with 'cookieHeader'
            if (srcLink) {
                const resSrc = await fetch(srcLink);
                const rawText = await resSrc.text();
                let newToken = "";
                
                try {
                    const jsonData = JSON.parse(rawText);
                    if (jsonData.cookieHeader) {
                        newToken = jsonData.cookieHeader.replace('__hdnea__=', '');
                    }
                } catch(e) {
                    // Fallback to Regex for plain text
                    newToken = (/__hdnea__=st=[^"'\s\r\n]*/.exec(rawText) || [""])[0].replace('__hdnea__=', '');
                }

                if (newToken) saveToken(newToken);
            }

            const resMain = await fetch(m3uMain);
            const rawData = await resMain.text();
            let finalChannels = [];

            if (rawData.trim().startsWith('{') || rawData.trim().startsWith('[')) {
                const jsonData = JSON.parse(rawData);
                const list = jsonData.channels || (Array.isArray(jsonData) ? jsonData : []);
                finalChannels = list.map(item => {
                    let extractedKeyId = null, extractedKey = null;
                    if (item.drm && typeof item.drm === 'object') {
                        const keys = Object.keys(item.drm);
                        if (keys.length > 0) {
                            extractedKeyId = keys[0];
                            extractedKey = item.drm[extractedKeyId];
                        }
                    }
                    return {
                        name: item.name || item.title || "Unknown",
                        logo: item.logo || item.image || item.tvg_logo || "",
                        url: item.mpd || item.url || item.link || "",
                        keyId: item.keyId || item.key_id || extractedKeyId,
                        key: item.key || item.license_key || extractedKey,
                        tokenValue: (item.token || item.url || "").match(/__hdnea__=([^&?]+)/)?.[1] || getToken()
                    };
                });
            } else {
                let m3uText = rawData;
                const savedToken = getToken();
                if (savedToken) m3uText = m3uText.replace(/(__hdnea__=)([^ \n\r\t"']*)/g, `__hdnea__=${savedToken}`);
                finalChannels = parseM3U(m3uText);
            }
            return finalChannels;
        } catch (e) { return []; }
    }

    function parseM3U(text) {
        const lines = text.split('\n').map(l => l.trim()).filter(Boolean);
        const channels = []; let cur = {};
        for (const line of lines) {
            if (line.startsWith("#EXTINF")) {
                cur.name = line.split(",").pop().trim();
                cur.logo = (line.match(/tvg-logo="(.*?)"/) || [])[1] || "";
            } else if (line.includes("license_key")) {
                const parts = line.split("=").pop().trim().split(":");
                if (parts.length === 2) [cur.keyId, cur.key] = parts;
            } else if (/^https?:\/\//i.test(line)) {
                cur.url = line;
                cur.tokenValue = line.match(/__hdnea__=([^&?]+)/)?.[1] || getToken();
                channels.push({ ...cur }); cur = {};
            }
        }
        return channels;
    }
